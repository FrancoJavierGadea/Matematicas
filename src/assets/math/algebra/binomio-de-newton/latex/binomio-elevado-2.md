$$
    ({\color{Blue} a} \pm {\color{Purple} b})^{2}
$$

$$
    ({\color{Blue} a} + {\color{Purple} b})^{2} = {\color{Blue} a}^{2} 
    \hspace{5pt} + \hspace{5pt} 2 \cdot {\color{Blue} a} \cdot {\color{Purple} b} 
    \hspace{5pt} + \hspace{5pt} {\color{Purple} b}^{2}
$$

$$
    ({\color{Blue} a} - {\color{Purple} b})^{2} = {\color{Blue} a}^{2} 
    \hspace{5pt} - \hspace{5pt} 2 \cdot {\color{Blue} a} \cdot {\color{Purple} b} 
    \hspace{5pt} + \hspace{5pt} {\color{Purple} b}^{2}
    $$
